{{/* since .Next/.Prev and .NextInSection/.PrevInSection can't be customized */}} {{/* to filter by, say, .File.Dir, let's write our own! */}} {{- $dirPages := where (where .Site.Pages "File.Dir" .Page.File.Dir) "Kind" "page" }} {{- $curPos := 0 }} {{- $idx := 0 }} {{- range $dirPages }} {{- if eq $ . }} {{- $curPos = $idx }} {{- end }} {{- $idx = add $idx 1}} {{- end }} {{- $nextIdx := sub $curPos 1 }} {{- $prevIdx := add $curPos 1 }} {{- $next := index $dirPages $nextIdx }} {{- $prev := index $dirPages $prevIdx }}
{{- with $next }} « {{ $next.Title | truncate 50 "..." }} {{ end }}
{{- with $prev }} {{ $prev.Title | truncate 50 "..."}} » {{ end }}
